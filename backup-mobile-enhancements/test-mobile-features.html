// 核心代码：一键拨号实现 function makePhoneCall(phoneNumber) { const cleanNumber = phoneNumber.replace(/[^\d+]/g, ''); if (isMobileDevice()) { // 移动端：使用tel协议 window.location.href = `tel:${cleanNumber}`; } else { // 桌面端：显示号码和复制选项 if (confirm(`拨打电话: ${phoneNumber}\n\n是否复制号码？`)) { navigator.clipboard.writeText(phoneNumber); } } } // 检测移动设备 function isMobileDevice() { return /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i .test(navigator.userAgent); }
