## 4. Nginx配置

### 4.1 当前配置状态
```nginx
# 当前配置 (v1生产 + v2测试)
location / {                    # v1生产版本
    root /home/admin/yunxinwangdian/frontend;
    index shop_index.html;
}

location /v2/ {                # v2测试版本
    alias /home/admin/yunxinwangdian/frontend-v2/;
    index index.html;
}
```

### 4.2 添加新版本配置 (v3)

#### 方法A: 手动编辑
```bash
# 1. 备份当前配置
sudo cp /etc/nginx/conf.d/yunxinwangdian.conf /etc/nginx/conf.d/yunxinwangdian.conf.backup-$(date +%Y%m%d)

# 2. 编辑配置文件
sudo vim /etc/nginx/conf.d/yunxinwangdian.conf

# 3. 在合适位置添加 (在v2配置后面)
location /v3/ {
    alias /home/admin/yunxinwangdian/frontend-v3/;
    index index.html;
}
```

#### 方法B: 使用sed命令
```bash
# 在v2配置后添加v3配置
sudo sed -i '/location \/v2\/ {/,/}/a\    # 单店版顾客端 v3 (新版本测试)\n    location /v3/ {\n        alias /home/admin/yunxinwangdian/frontend-v3/;\n        index index.html;\n    }' /etc/nginx/conf.d/yunxinwangdian.conf
```

### 4.3 应用配置
```bash
# 1. 测试配置语法
sudo nginx -t
# 应该输出: syntax is ok, test is successful

# 2. 重载Nginx (不中断服务)
sudo systemctl reload nginx

# 3. 检查服务状态
sudo systemctl status nginx
```

### 5.2 详细验证
```bash
# 创建验证脚本
cat > /tmp/verify-deploy.sh << 'EOF'
#!/bin/bash
echo "=== 部署验证报告 ==="
echo "时间: $(date)"
echo ""

# 检查HTTP状态
check_url() {
    local url=$1
    local name=$2
    local status=$(curl -s -o /dev/null -w "%{http_code}" "$url")
    local size=$(curl -s "$url" | wc -c)
    
    if [ "$status" = "200" ]; then
        echo "✅ $name: HTTP $status, 大小: $size 字节"
        return 0
    else
        echo "❌ $name: HTTP $status"
        return 1
    fi
}

# 验证各版本

check_url "http://localhost/v2/" "v2测试版本"
check_url "http://localhost/v3/" "v3新版本"

EOF
```

## 6. 切换到生产

### 6.1 准备工作
```bash
# 1. 确保v3版本通过测试
# 2. 备份当前生产配置
sudo cp /etc/nginx/conf.d/yunxinwangdian.conf /etc/nginx/conf.d/yunxinwangdian.conf.backup-before-v3

# 3. 创建切换脚本
cat > /tmp/switch-to-v3.sh << 'EOF'
#!/bin/bash
set -e
echo "🔄 开始切换到v3生产环境..."

# 备份时间
BACKUP_TIME=$(date +%Y%m%d-%H%M%S)

# 备份当前配置
sudo cp /etc/nginx/conf.d/yunxinwangdian.conf /etc/nginx/conf.d/yunxinwangdian.conf.backup-$BACKUP_TIME

# 修改配置：v3成为生产版本，v1移到/v1/路径
sudo sed -i 's|location / {|location /v1/ {\n        alias /home/admin/yunxinwangdian/frontend/;\n        index shop_index.html;\n    }\n\n    # 单店版顾客端 v3 (新生产版本)\n    location / {|' /etc/nginx/conf.d/yunxinwangdian.conf

# 修改v3配置：从alias改为root
sudo sed -i 's|location /v3/ {|location / {|' /etc/nginx/conf.d/yunxinwangdian.conf
sudo sed -i 's|alias /home/admin/yunxinwangdian/frontend-v3/;|root /home/admin/yunxinwangdian/frontend-v3;|' /etc/nginx/conf.d/yunxinwangdian.conf

# 测试配置
sudo nginx -t || {
    echo "❌ 配置错误，恢复备份"
    sudo cp /etc/nginx/conf.d/yunxinwangdian.conf.backup-$BACKUP_TIME /etc/nginx/conf.d/yunxinwangdian.conf
    exit 1
}

# 重载Nginx
sudo systemctl reload nginx

echo "✅ 切换完成！"
echo "生产版本: v3 (http://localhost/)"
echo "旧版本: v1 (http://localhost/v1/)"
echo "备份文件: yunxinwangdian.conf.backup-$BACKUP_TIME"
EOF

chmod +x /tmp/switch-to-v3.sh
```

### 6.2 执行切换
```bash
# 1. 再次验证v3
curl -f http://localhost/v3/ || { echo "v3测试失败"; exit 1; }

# 2. 执行切换
sudo /tmp/switch-to-v3.sh

# 3. 验证切换结果
curl -f http://localhost/ && echo "✅ v3已成为生产版本"
curl -f http://localhost/v1/ && echo "✅ v1仍可访问"
```

## 7. 回滚操作

### 7.1 快速回滚到v1
```bash
# 回滚脚本
cat > /tmp/rollback-to-v1.sh << 'EOF'
#!/bin/bash
echo "🔄 开始回滚到v1..."

# 查找最新的备份文件
BACKUP_FILE=$(ls -t /etc/nginx/conf.d/yunxinwangdian.conf.backup-* 2>/dev/null | head -1)

if [ -z "$BACKUP_FILE" ]; then
    echo "❌ 找不到备份文件"
    exit 1
fi

echo "使用备份文件: $BACKUP_FILE"

# 恢复配置
sudo cp "$BACKUP_FILE" /etc/nginx/conf.d/yunxinwangdian.conf

# 测试配置
sudo nginx -t || {
    echo "❌ 配置错误"
    exit 1
}

# 重载Nginx
sudo systemctl reload nginx

echo "✅ 回滚完成"
echo "生产版本已恢复为v1"
EOF

chmod +x /tmp/rollback-to-v1.sh
```

### 7.2 手动回滚步骤
```bash
# 1. 停止Nginx (可选，紧急时使用)

# 4. 启动/重载Nginx
```

```bash
# 创建监控脚本
cat > /tmp/monitor-services.sh << 'EOF'
#!/bin/bash
echo "=== 服务监控报告 ==="
echo "时间: $(date)"

    echo "  $line"

ps aux | grep -E "nginx|node" | grep -v grep | awk '{print $4, $11}' | sort -rn | head -5
```

## 9. 故障处理

### 9.1 常见问题及解决

#### 问题1: Nginx配置错误
```bash
症状: sudo nginx -t 报错
解决:
1. 检查语法错误
   sudo nginx -t 2>&1 | grep -A2 -B2 "error"
   
2. 恢复最近备份
   sudo cp $(ls -t /etc/nginx/conf.d/yunxinwangdian.conf.backup-* | head -1) /etc/nginx/conf.d/yunxinwangdian.conf
   
3. 手动修复
   sudo vim /etc/nginx/conf.d/yunxinwangdian.conf
```

#### 问题2: 网站返回404
```bash
症状: curl返回404
解决:
1. 检查文件是否存在
   ls -la /home/admin/yunxinwangdian/frontend*/index.html
   
2. 检查Nginx配置路径
   grep -n "root\|alias" /etc/nginx/conf.d/yunxinwangdian.conf
   
3. 检查文件权限
   ls -la /home/admin/yunxinwangdian/frontend-v3/
   # 确保Nginx用户有读取权限
```

#### 问题3: 502 Bad Gateway
```bash
症状: API返回502
解决:
1. 检查后端服务
   ps aux | grep node
   
2. 重启后端服务
   cd /home/admin/yunxinwangdian/backend
   pkill -f "node server.js"
   nohup node server.js > ../logs/app.log 2>&1 &
   
3. 检查端口
   netstat -tlnp | grep 3000
```

#### 问题4: 权限问题
```bash
症状: 403 Forbidden
解决:
1. 检查目录权限
   sudo ls -la /home/admin/yunxinwangdian/
   
2. 修复权限
   sudo chmod -R 755 /home/admin/yunxinwangdian/frontend-v3/
   sudo chown -R admin:admin /home/admin/yunxinwangdian/frontend-v3/
```

### 9.2 紧急恢复流程
```bash
# 紧急恢复脚本
cat > /tmp/emergency-recovery.sh << 'EOF'
#!/bin/bash
echo "🚨 紧急恢复流程启动"

echo "4. 验证恢复..."
sleep 2
if curl -s -o /dev/null -w "%{http_code}" http://localhost/ | grep -q "200"; then
    echo "✅ 恢复成功"
else
    echo "❌ 恢复失败，需要手动检查"
    exit 1
fi
EOF

chmod +x /tmp/emergency-recovery.sh
```

```bash
# 3. 测试配置
```
