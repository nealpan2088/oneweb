## 🔧 实用修改脚本

### **脚本1：快速替换文本**
```bash
#!/bin/bash
# replace-text.sh
OLD_TEXT="$1"
NEW_TEXT="$2"
FILE="/home/admin/yunxinwangdian/frontend/index.html"

if [ -z "$OLD_TEXT" ] || [ -z "$NEW_TEXT" ]; then
    echo "用法: $0 \"旧文本\" \"新文本\""
    echo "示例: $0 \"8年开发经验\" \"10年全栈经验\""
    exit 1
fi

# 备份
cp "$FILE" "$FILE.backup-$(date +%Y%m%d-%H%M%S)"

# 替换
sed -i "s|$OLD_TEXT|$NEW_TEXT|g" "$FILE"

echo "✅ 替换完成"
echo "旧: $OLD_TEXT"
echo "新: $NEW_TEXT"
echo "备份: $FILE.backup-$(date +%Y%m%d-%H%M%S)"
```

### **脚本2：修改DEMO项目**
```bash
#!/bin/bash
# update-demo.sh
DEMO_NUM=$1
NEW_TITLE="$2"
NEW_DESC="$3"
NEW_LINK="$4"

FILE="/home/admin/yunxinwangdian/frontend/index.html"

if [ -z "$DEMO_NUM" ]; then
    echo "用法: $0 <DEMO编号> \"新标题\" \"新描述\" \"新链接\""
    echo "DEMO编号: 1-6 (对应页面上的6个DEMO)"
    echo "示例: $0 1 \"智能电商\" \"电商平台解决方案\" \"https://example.com\""
    exit 1
fi

# 根据DEMO编号计算行数范围
case $DEMO_NUM in
    1) START=280; END=320 ;;
    2) START=330; END=370 ;;
    3) START=380; END=420 ;;
    4) START=430; END=470 ;;
    5) START=480; END=520 ;;
    6) START=530; END=570 ;;
    *) echo "错误: DEMO编号必须是1-6"; exit 1 ;;
esac

# 备份
cp "$FILE" "$FILE.backup-demo$DEMO_NUM"

# 修改标题
if [ -n "$NEW_TITLE" ]; then
    sed -i "${START},${END}s|<h3 class=\"demo-title\">[^<]*</h3>|<h3 class=\"demo-title\">$NEW_TITLE</h3>|" "$FILE"
fi

# 修改描述
if [ -n "$NEW_DESC" ]; then
    sed -i "${START},${END}s|<p class=\"demo-description\">[^<]*</p>|<p class=\"demo-description\">$NEW_DESC</p>|" "$FILE"
fi

# 修改链接
if [ -n "$NEW_LINK" ]; then
    sed -i "${START},${END}s|href=\"[^\"]*\"|href=\"$NEW_LINK\"|" "$FILE"
fi

echo "✅ DEMO $DEMO_NUM 修改完成"
```
