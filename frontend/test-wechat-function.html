4. 技术实现说明

改进的微信功能实现：

async function showWechatInfo() {
    try {
        // 1. 从API获取店铺信息
        const shopInfo = await shopService.getShopInfo();
        
        if (shopInfo && shopInfo.wechat_id) {
            // 2. 个性化提示信息
            const message = `💬 联系${shopInfo.name}微信\n\n` +
                           `微信号: ${shopInfo.wechat_id}\n\n` +
                           `操作步骤说明...`;
            
            alert(message);
            
            // 3. 记录访问统计
            await shopService.recordVisit('click_wechat');
            
            // 4. 提供复制功能
            if (confirm('复制微信号？')) {
                navigator.clipboard.writeText(shopInfo.wechat_id);
            }
        } else {
            // 5. 友好的默认信息
            alert('💬 联系店铺微信\n\n微信号: WangCafe_Service\n\n...');
        }
    } catch (error) {
        // 6. 错误处理
        console.error('显示微信信息失败:', error);
        alert('💬 联系店铺微信\n\n微信号: WangCafe_Service\n\n...');
    }
}
